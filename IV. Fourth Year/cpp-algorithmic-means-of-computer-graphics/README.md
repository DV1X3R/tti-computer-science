import math as m

def t(Dx, Dy):
    return [[1, 0, 0], [0, 1, 0], [Dx, Dy, 1]]

def s(Sx, Sy):
    return [[Sx, 0, 0], [0, Sy, 0], [0, 0, 1]]

def r(O):
    o = m.radians(O)
    return [[m.cos(o), m.sin(o), 0], [-m.sin(o), m.cos(o), 0], [0, 0, 1]]

def sr(src, X0, Y0, Sx, Sy, O):
    p = np.matmul(t(-X0, -Y0), s(Sx, Sy))
    p = np.matmul(p, r(O))
    p = np.matmul(p, t(X0, Y0))
    return np.matmul(src, p)
